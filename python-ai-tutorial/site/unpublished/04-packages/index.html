    Now that you understand modules, let's explore packages - collections of modules that others have created and shared!

    What is PyPI?¶

    PyPI (Python Package Index) is like an app store for Python code: - üì¶ 400,000+ packages available - üåç Created by developers worldwide - üÜì Free to use!

    Visit pypi.org to explore!

    Installing Your First Package¶

    We'll use rich - a package that makes terminal output beautiful!

    In your terminal:

    pip install rich

    What's pip?

    pip is Python's package installer - like downloading apps on your phone!

    Rich Panels and Styling¶

    Complete Rich Chatbot¶

    Let's combine everything into an impressive chatbot:

    # rich_ai_chatbot.py
from rich.console import Console
from rich.panel import Panel
from rich.prompt import Prompt, Confirm
from rich.table import Table
from rich.progress import Progress, SpinnerColumn, TextColumn
from rich.markdown import Markdown
import time
import random

console = Console()

class RichChatBot:
    def __init__(self):
        self.name = "RichBot 3000"
        self.conversations = 0

    def welcome(self):
        """Display welcome message"""
        welcome_text = f"""
# {self.name} ü§ñ

Welcome to the **future** of chatbots!

I can:
- Chat with style ‚ú®
- Tell jokes üòÑ
- Show information beautifully üìä
- Think before I speak ü§î
        """
        md = Markdown(welcome_text)
        console.print(Panel(md, border_style="bright_blue"))

    def think(self, duration=1):
        """Show thinking animation"""
        with Progress(
            SpinnerColumn(),
            TextColumn("[progress.description]{task.description}"),
            console=console,
            transient=True,
        ) as progress:
            progress.add_task(description="Processing...", total=None)
            time.sleep(duration)

    def respond(self, user_input):
        """Generate styled response"""
        self.conversations += 1

        # Simulate thinking
        self.think(0.5)

        responses = {
            "hello": "Hello there! How can I brighten your day? ‚òÄÔ∏è",
            "joke": self.tell_joke(),
            "help": self.show_help(),
            "stats": f"We've had {self.conversations} exchanges so far!",
        }

        # Get response or default
        response = responses.get(
            user_input.lower(), 
            f"Interesting! Tell me more about '{user_input}'"
        )

        # Display in panel
        console.print(Panel(
            response,
            title=f"üí¨ {self.name}",
            border_style="green"
        ))

    def tell_joke(self):
        """Return a formatted joke"""
        jokes = [
            "Why do programmers prefer dark mode?\n[italic]Because light attracts bugs! üêõ[/italic]",
            "What's a robot's favorite music?\n[italic]Heavy metal! üé∏[/italic]",
            "Why did the AI go to therapy?\n[italic]It had too many deep issues! üß†[/italic]"
        ]
        return random.choice(jokes)

    def show_help(self):
        """Return help table"""
        return "Try: 'hello', 'joke', 'stats', or ask me anything!"

    def run(self):
        """Main chat loop"""
        self.welcome()

        user_name = Prompt.ask("\n[bold cyan]What's your name?[/bold cyan]")
        console.print(f"\nNice to meet you, [bold]{user_name}[/bold]! üëã\n")

        while True:
            user_input = Prompt.ask("[bold]You[/bold]")

            if user_input.lower() in ['bye', 'exit', 'quit']:
                if Confirm.ask("Are you sure you want to leave?"):
                    farewell = Panel(
                        f"Goodbye {user_name}! Thanks for chatting! üëã\n"
                        f"We had {self.conversations} exchanges.",
                        title="üëã Farewell",
                        border_style="yellow"
                    )
                    console.print(farewell)
                    break
            else:
                self.respond(user_input)

# Run the chatbot
if __name__ == "__main__":
    bot = RichChatBot()
    bot.run()

    Package Management Best Practices¶

    Creating requirements.txt¶

    Save your project's packages:

    pip freeze > requirements.txt

    Your requirements.txt will look like: 

    rich==13.7.0
colorama==0.4.6
termcolor==2.3.0
pyfiglet==0.8.post1

    Installing from requirements.txt¶

    Others can install your packages:

    pip install -r requirements.txt

    Challenge: Ultimate Pretty Bot üåü¶

    Create a chatbot that uses: 1. Rich for beautiful output 2. ASCII art for the title 3. Loading animations 4. A help menu table

    üîç Solution 
    # ultimate_pretty_bot.py
from rich.console import Console
from rich.panel import Panel
from rich.table import Table
from rich.progress import Progress, SpinnerColumn, TextColumn
from rich.prompt import Prompt
import pyfiglet
import time
import random

console = Console()

class UltimatePrettyBot:
    def __init__(self):
        self.commands = {
            "help": "Show available commands",
            "joke": "Tell a random joke",
            "fact": "Share an AI fact",
            "ascii": "Show ASCII art",
            "about": "About this bot"
        }

    def show_banner(self):
        """Display ASCII banner"""
        banner = pyfiglet.figlet_format("ULTRA BOT", font="slant")
        console.print(f"[bold magenta]{banner}[/bold magenta]")
        console.print("[italic cyan]The Ultimate Pretty ChatBot Experience[/italic cyan]\n", justify="center")

    def show_help(self):
        """Display help table"""
        table = Table(title="üéÆ Available Commands", show_header=True)
        table.add_column("Command", style="cyan", width=12)
        table.add_column("Description", style="yellow")

        for cmd, desc in self.commands.items():
            table.add_row(f"/{cmd}", desc)

        return table

    def process_command(self, command):
        """Process user commands"""
        with Progress(
            SpinnerColumn(spinner_name="dots12"),
            TextColumn("[bold blue]Thinking...[/bold blue]"),
            transient=True,
            console=console
        ) as progress:
            progress.add_task("think", total=None)
            time.sleep(1)

        if command == "/help":
            return self.show_help()
        elif command == "/joke":
            jokes = [
                "Why don't AIs ever get lonely?\n[green]They always have their algorithms![/green]",
                "What do you call a neural network that sings?\n[green]A Dell![/green]"
            ]
            return Panel(random.choice(jokes), title="üòÑ AI Joke", border_style="yellow")
        elif command == "/fact":
            facts = [
                "The first AI program was written in 1951!",
                "GPT-3 has 175 billion parameters!"
            ]
            return Panel(random.choice(facts), title="üß† AI Fact", border_style="cyan")
        elif command == "/ascii":
            art = pyfiglet.figlet_format("HELLO!", font="bubble")
            return Panel(f"[bold green]{art}[/bold green]", title="üé® ASCII Art")
        elif command == "/about":
            return Panel(
                "[bold]UltraBot v1.0[/bold]\n\n"
                "Created with ‚ù§Ô∏è using Python and Rich\n"
                "The prettiest chatbot you'll ever meet!",
                title="‚ÑπÔ∏è About",
                border_style="blue"
            )
        else:
            return Panel(
                f"I don't understand '{command}'. Try /help!",
                title="‚ùì Unknown Command",
                border_style="red"
            )

    def run(self):
        """Main bot loop"""
        console.clear()
        self.show_banner()

        # Welcome panel
        welcome = Panel(
            "Welcome to UltraBot! üöÄ\n\n"
            "Type [bold cyan]/help[/bold cyan] to see what I can do!\n"
            "Type [bold red]exit[/bold red] to leave.",
            title="üëã Welcome",
            border_style="bright_blue"
        )
        console.print(welcome)

        while True:
            user_input = Prompt.ask("\n[bold white]You[/bold white]")

            if user_input.lower() == "exit":
                goodbye = Panel(
                    "Thanks for chatting! See you next time! üëã",
                    title="Goodbye",
                    border_style="yellow"
                )
                console.print(goodbye)
                break

            result = self.process_command(user_input.lower())
            console.print(result)

# Run the bot
if __name__ == "__main__":
    bot = UltimatePrettyBot()
    bot.run()
