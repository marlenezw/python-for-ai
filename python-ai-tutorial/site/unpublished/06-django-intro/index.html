    Models - Storing Chat Data¶

    Create chatbot/models.py:

    # chatbot/models.py
from django.db import models
from django.utils import timezone

class Chat(models.Model):
    """A chat session"""
    created_at = models.DateTimeField(default=timezone.now)
    title = models.CharField(max_length=100, default="New Chat")

    def __str__(self):
        return f"{self.title} - {self.created_at.strftime('%Y-%m-%d %H:%M')}"

    class Meta:
        ordering = ['-created_at']

class Message(models.Model):
    """A message in a chat"""
    chat = models.ForeignKey(Chat, on_delete=models.CASCADE, related_name='messages')
    content = models.TextField()
    is_user = models.BooleanField(default=True)
    created_at = models.DateTimeField(default=timezone.now)

    def __str__(self):
        sender = "User" if self.is_user else "AI"
        return f"{sender}: {self.content[:50]}..."

    class Meta:
        ordering = ['created_at']

    Admin Interface¶

    Register your models in chatbot/admin.py:

    # chatbot/admin.py
from django.contrib import admin
from .models import Chat, Message

@admin.register(Chat)
class ChatAdmin(admin.ModelAdmin):
    list_display = ['title', 'created_at']
    list_filter = ['created_at']
    search_fields = ['title']

@admin.register(Message)
class MessageAdmin(admin.ModelAdmin):
    list_display = ['chat', 'is_user', 'content_preview', 'created_at']
    list_filter = ['is_user', 'created_at']
    search_fields = ['content']

    def content_preview(self, obj):
        return obj.content[:50] + "..." if len(obj.content) > 50 else obj.content
    content_preview.short_description = 'Content'

    Create a superuser to access admin:

    python manage.py createsuperuser

    Understanding Django Flow¶

    Create a visual representation:

    # django_flow.py
"""
Django Request Flow:

1. User visits URL (e.g., /chat)
   ‚Üì
2. Django checks urls.py for matching pattern
   ‚Üì
3. Calls the associated view function
   ‚Üì
4. View processes request (maybe uses models)
   ‚Üì
5. View renders template with context data
   ‚Üì
6. HTML response sent to user

Example:
User ‚Üí /chat ‚Üí urls.py ‚Üí views.chat() ‚Üí chat.html ‚Üí User sees page
"""

    Challenge: Create a Simple View üéØ¶

    Create a new view that: 1. Shows the current time 2. Counts total messages 3. Displays a random AI fact

    üîç Solution 
    # Add to chatbot/views.py
from django.shortcuts import render
from django.utils import timezone
from .models import Message
import random

def stats(request):
    """Show chat statistics"""
    ai_facts = [
        "The first chatbot ELIZA was created in 1966",
        "GPT-3 has 175 billion parameters",
        "AI can now generate images from text descriptions",
        "Machine learning was coined in 1959",
    ]

    context = {
        'current_time': timezone.now(),
        'total_messages': Message.objects.count(),
        'user_messages': Message.objects.filter(is_user=True).count(),
        'ai_messages': Message.objects.filter(is_user=False).count(),
        'random_fact': random.choice(ai_facts),
    }

    return render(request, 'chatbot/stats.html', context)
